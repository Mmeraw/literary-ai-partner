/**
 * GOVERNANCE VERSION SERVICE
 * Standalone function that provides governance constants
 * Internal service - no auth required
 */

const GOVERNANCE_VERSION = "1.0.0";

const CANON_HASHES = {
    EVALUATE_ENTRY_CANON: "EVALUATE_ENTRY_CANON_v1.2",
    WAVE_GUIDE: "WAVE_GUIDE_v2.1",
    SYNOPSIS_SPEC: "SYNOPSIS_SPEC_v1.0",
    QUERY_LETTER_SPEC: "QUERY_LETTER_SPEC_v1.0",
    SCREENPLAY_QUALITY_STANDARD: "SCREENPLAY_QUALITY_STANDARD_v1.0",
    AGENT_PACKAGE_SPEC: "AGENT_PACKAGE_SPEC_v1.0",
    PITCH_DECK_SPEC: "PITCH_DECK_SPEC_v1.0",
    COMPARABLES_SPEC: "COMPARABLES_SPEC_v1.0"
};

Deno.serve(async (req) => {
    try {
        const { action, ...params } = await req.json();

        if (action === 'getVersion') {
            return Response.json({ governanceVersion: GOVERNANCE_VERSION });
        }

        if (action === 'getCanonHash') {
            const { canonName } = params;
            return Response.json({ 
                canonHash: CANON_HASHES[canonName] || null 
            });
        }

        if (action === 'buildRefusalResponse') {
            const { status, code, userMessage, developerMessage, refusalReason, nextAction } = params;
            return Response.json({
                status: status || "blocked",
                code,
                user_message: userMessage,
                developer_message: developerMessage || null,
                refusal_reason: refusalReason,
                next_action: nextAction || "none"
            });
        }

        if (action === 'buildAuditBase') {
            const { eventName, functionId, canonHash, userEmail, requestId } = params;
            return Response.json({
                event_id: `evt_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
                request_id: requestId || `req_${Date.now()}`,
                timestamp_utc: new Date().toISOString(),
                function_id: functionId,
                canon_hash: canonHash,
                governance_version: GOVERNANCE_VERSION,
                user_email: userEmail
            });
        }

        return Response.json({ error: 'Invalid action' }, { status: 400 });

    } catch (error) {
        return Response.json({ error: error.message }, { status: 500 });
    }
});
