/**
 * PHASE 3 FUNCTION #4: rejectAgentVerification
 * 
 * Authority: AGENT_ONBOARDING_VERIFICATION_SPEC_v1.0.0.md
 * State Transition: PENDING → REJECTED (admin-only)
 * 
 * Release-Blocking Tests:
 * 1. Non-admin denied (403)
 * 2. State machine (only PENDING→REJECTED allowed)
 * 3. No side effects (no roles, no org, no access)
 * 4. Admin DTO keyset exact
 * 5. Safe error shape ({ code, message, requestId })
 * 6. Audit event verification_rejected emitted
 */

import { createClientFromRequest } from 'npm:@base44/sdk@0.8.6';

const CANON_VERSION = 'AGENT_ONBOARDING_VERIFICATION_SPEC_v1.0.0';

/**
 * Admin Decision DTO (ADMIN_VERIFICATION_DECISION_DTO_v1.0.0)
 * Admin-only verification decision response
 */
function toAdminVerificationDecisionDTO(targetUserId, previousState, newState, requestId, updatedAt) {
    return {
        targetUserId,
        previousState,
        newState,
        requestId,
        updatedAt
    };
}

/**
 * Safe error response (no stack traces, no internal IDs)
 * Canon shape: { code, message, requestId } ONLY
 */
function errorResponse(code, message, requestId, status = 400) {
    return Response.json({
        code,
        message,
        requestId
    }, { status });
}

Deno.serve(async (req) => {
    const requestId = `req_${Date.now()}_${Math.random().toString(36).slice(2, 9)}`;
    
    try {
        const base44 = createClientFromRequest(req);
        const user = await base44.auth.me();
        
        if (!user) {
            return errorResponse('UNAUTHORIZED', 'Authentication required', requestId, 401);
        }

        // ADMIN GATE: Only admins can reject verification requests
        if (user.role !== 'admin') {
            return errorResponse(
                'ADMIN_REQUIRED',
                'Only administrators can reject verification requests',
                requestId,
                403
            );
        }

        const { industry_user_id } = await req.json();

        // Validation: Required field
        if (!industry_user_id) {
            return errorResponse(
                'VALIDATION_FAILED',
                'Missing required field: industry_user_id',
                requestId
            );
        }

        // Fetch the IndustryUser record
        const records = await base44.asServiceRole.entities.IndustryUser.filter({
            id: industry_user_id
        });

        if (records.length === 0) {
            return errorResponse(
                'NOT_FOUND',
                'Industry user record not found',
                requestId,
                404
            );
        }

        const industryUser = records[0];

        // STATE MACHINE: Only PENDING → REJECTED transition allowed
        if (industryUser.verification_status !== 'PENDING') {
            return errorResponse(
                'STATE_VIOLATION',
                `Cannot reject from status: ${industryUser.verification_status}. Only PENDING requests can be rejected.`,
                requestId
            );
        }

        const previousState = industryUser.verification_status;
        const newState = 'REJECTED';
        const updatedAt = new Date().toISOString();

        // Transition: PENDING → REJECTED (no role grants, no org changes, no messaging)
        await base44.asServiceRole.entities.IndustryUser.update(industryUser.id, {
            verification_status: newState
        });

        // AUDIT LOGGING: Emit append-only audit event
        await base44.asServiceRole.entities.EvaluationAuditEvent.create({
            event_type: 'verification_rejected',
            entity_type: 'IndustryUser',
            entity_id: industryUser.id,
            user_email: user.email,
            metadata: {
                target_user_email: industryUser.user_email,
                previous_state: previousState,
                new_state: newState,
                rejected_by: user.email,
                request_id: requestId
            }
        });

        return Response.json(
            toAdminVerificationDecisionDTO(
                industryUser.user_email,
                previousState,
                newState,
                requestId,
                updatedAt
            )
        );

    } catch (error) {
        console.error('rejectAgentVerification error:', error);
        return errorResponse(
            'INTERNAL_ERROR',
            'Failed to reject verification request',
            requestId,
            500
        );
    }
});
